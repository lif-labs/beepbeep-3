374  public static boolean parseBoolValue(Object o)
375  {
376  if (o instanceof Boolean)
377  {
378  return (Boolean) o;
379  }
380  else if (o instanceof String)
381  {
382  String s = (String) o;
383  return s.compareToIgnoreCase("true") == 0 || s.compareToIgnoreCase("T") == 0
384  || s.compareToIgnoreCase("1") == 0;
385  }
386  if (o instanceof Number)
387  {
388  Number n = (Number) o;
389  return Math.abs(n.doubleValue()) >= 0.00001;
390  }
391  // When in doubt, return false
