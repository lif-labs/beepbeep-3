173  @Override
174  public NextStatus hasNextSoft()
175  {
176  if (!m_outputQueues[0].isEmpty())
177  {
178  return NextStatus.YES;
179  }
180  boolean all_no = true;
181  NextStatus out = NextStatus.MAYBE;
182  for (Pullable p : m_inputPullables)
183  {
184  NextStatus ns = p.hasNextSoft();
185  if (ns != NextStatus.NO)
186  {
187  all_no = false;
188  }
189  if (ns == NextStatus.YES)
190  {
191  // We don't do a "break" here.
192  // We must go through ALL pullables, even if we encounter one
193  // that says yes. Otherwise, we might end up pulling events from
194  // the same pullable all the time.
195  out = NextStatus.YES;
196  }
197  }
198  if (all_no)
199  {
200  return NextStatus.NO;
201  }
202  return out;
203  }

205  @Override

207  {
208  if (!m_outputQueues[0].isEmpty())
209  {
210  return true;
211  }
212  boolean all_no = true;
213  NextStatus out = NextStatus.MAYBE;
214  for (int i = 0; i < Processor.MAX_PULL_RETRIES; i++)
215  {
216  for (Pullable p : m_inputPullables)
217  {
218  boolean ns = p.hasNext();
219  if (ns)
220  {
221  all_no = false;
222  // We don't do a "break" here.
223  // We must go through ALL pullables, even if we encounter one
224  // that says yes. Otherwise, we might end up pulling events from
225  // the same pullable all the time.
226  out = NextStatus.YES;
227  }
228  }
229  if (all_no)

232  }

236  }
237  }
