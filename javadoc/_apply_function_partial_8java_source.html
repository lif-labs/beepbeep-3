131  @Override
132  public Pushable push(Object o)
133  {
134  Object[] out_front = new Object[m_outputPullables.length];
135  if (m_frontNumber[m_index] < m_numCurrentFront)
136  {
137  // This is an event from a front that has already
138  // been processed. Don't bother putting it in the
139  // queue or evaluate the function.
140  m_frontNumber[m_index]++;
141  return this;
142  }
143  assert m_frontNumber[m_index] >= m_numCurrentFront;
144  m_inputQueues[m_index].add(o);
145  boolean evaluated = true;
146  // the following is actually a bounded while(evaluated)
147  for (int loop_count = 0; loop_count < MAX_LOOPS && evaluated; loop_count++)
148  {
149  for (int i = 0; i < m_inputPullables.length; i++)
150  {
151  if (m_frontNumber[i] == m_numCurrentFront)
152  {
153  if (!m_inputQueues[i].isEmpty())
154  {
155  m_inputFront[i] = m_inputQueues[i].remove();
156  m_frontNumber[i]++;
157  }
158  else
159  {
160  m_inputFront[i] = null;
161  }
162  }
163  }
164  evaluated = m_function.evaluatePartial(m_inputFront, out_front, m_context);
165  if (evaluated)
166  {
167  for (int i = 0; i < out_front.length; i++)
168  {
169  m_outputPushables[i].push(out_front[i]);
170  }
171  for (int i = 0; i < m_inputFront.length; i++)
172  {
173  m_inputFront[i] = null;
174  }
175  m_numCurrentFront++;
176  }
177  }
178  return this;
179  }
