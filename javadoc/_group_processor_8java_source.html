369  public synchronized Map<Integer, Processor> cloneInto(GroupProcessor group, boolean with_state)
370  {
371  super.duplicateInto(group);
372  if (group.m_eventTracker != null)
373  {
374  group.m_eventTracker.add(group);
375  }
376  group.m_notifySources = m_notifySources;
377  Map<Integer, Processor> new_procs = new HashMap<Integer, Processor>();
378  EventTracker new_tracker = null;
379  if (m_innerTracker != null)
380  {
381  new_tracker = m_innerTracker.getCopy(false);
382  }
383  group.m_innerTracker = new_tracker;
384  Processor start = null;
385  // Clone every processor of the original group
386  for (Processor p : m_processors)
387  {
388  if (start == null && p.getOutputArity() > 0)
389  {
390  start = p;
391  }
392  Processor clone_p = copyProcessor(p, with_state);
393  new_procs.put(p.getId(), clone_p);
394  group.addProcessor(clone_p);
395  }
396  // Re-pipe the inputs and outputs like in the original group
397  associateEndpoints(group, new_procs);
398  // Re-pipe the internal processors like in the original group
399  CopyCrawler cc = new CopyCrawler(new_procs, new_tracker);
400  cc.crawl(start);
401  return new_procs;
402  }

413  protected synchronized void associateEndpoints(GroupProcessor group,
414  Map<Integer, Processor> new_procs)
415  {
416  // Re-pipe the inputs like in the original group
417  for (Map.Entry<Integer, ProcessorAssociation> entry : m_inputPullableAssociations.entrySet())
418  {
419  int input_number = entry.getKey();
420  ProcessorAssociation pa = entry.getValue();
421  Processor clone_p = new_procs.get(pa.m_processor.getId());
422  group.associateInput(input_number, clone_p, pa.m_ioNumber);
423  }
424  // Re-pipe the outputs like in the original group
425  for (Map.Entry<Integer, ProcessorAssociation> entry : m_outputPushableAssociations.entrySet())
426  {
427  int output_number = entry.getKey();
428  ProcessorAssociation pa = entry.getValue();
429  Processor clone_p = new_procs.get(pa.m_processor.getId());
430  group.associateOutput(output_number, clone_p, pa.m_ioNumber);
431  }

446  {

448  clone_p.setContext(p.m_context);
449  if (with_state)

457  {

507  @Override
508  public synchronized void visit(Processor p)
509  {
510  int out_arity = p.getOutputArity();
511  for (int i = 0; i < out_arity; i++)
512  {
513  Pushable push = p.getPushableOutput(i);
514  if (push != null)
515  {
516  Processor target = push.getProcessor();
517  int j = push.getPosition();
518  Processor new_p;
519  Processor new_target;
520  synchronized (m_correspondences)
521  {
522  new_p = m_correspondences.get(p.getId());
523  new_target = m_correspondences.get(target.getId());
524  }
525  if (new_p != null && new_target != null)
526  {
527  // new_p and new_target may be null if they refer to a processor
528  // outside of the group
529  Connector.connect(m_tracker, new_p, i, new_target, j);
530  }
531  }
532  }

710  m_hasBeenNotifiedOfEndOfTrace[m_position] = true;
711  if (!allNotifiedEndOfTrace())
712  {

714  }
715  // Notify the end of trace on all the inner Pushables

722  Queue<Object[]> temp_queue = new ArrayDeque<Object[]>();

755  private final void outputEvent(Queue<Object[]> temp_queue, boolean outs)
756  {
757  if (outs && !temp_queue.isEmpty())
758  {
759  for (Object[] evt : temp_queue)
760  {

954  @SuppressWarnings("unchecked")
955  @Override
956  public GroupProcessor readState(Object o) throws ProcessorException
957  {
958  Map<String,Object> contents = (HashMap<String,Object>) o;
959  // Create processor
960  GroupProcessor gp = new GroupProcessor(((Number) contents.get("in-arity")).intValue(),
961  ((Number) contents.get("out-arity")).intValue());
962  // Add internal processors (regular and source)
963  Map<Integer,Processor> procs = new HashMap<Integer,Processor>();
964  for (Processor p : (List<Processor>) contents.get("processors"))
965  {
966  gp.addProcessor(p);
967  procs.put(p.getId(), p);
968  }
969  for (Processor p : (Set<Processor>) contents.get("sources"))
970  {
971  gp.addProcessor(p);
972  procs.put(p.getId(), p);
973  }
974  // Connect each of them according to the connections
975  Set<Connector.Connection> connections = (Set<Connector.Connection>) contents.get("connections");
976  for (Connector.Connection conn : connections)
977  {
978  Processor p1 = procs.get(conn.m_sourceProcessorId);
979  Processor p2 = procs.get(conn.m_destinationProcessorId);
980  if (p1 == null || p2 == null)
981  {
982  // Ignore
983  continue;
984  }
985  Connector.connect(p1, conn.m_sourcePipeNumber, p2, conn.m_destinationPipeNumber);
986  }
987  // Associate group's inputs to processor inputs
988  for (List<Integer> list : ((Set<List<Integer>>) contents.get("input-associations")))
989  {
990  int io_1 = list.get(0);
991  int io_2 = list.get(1);
992  int p_id = list.get(2);
993  if (!procs.containsKey(p_id))
994  {
995  // Ignore
996  continue;
997  }
998  Processor p = procs.get(p_id);
999  gp.associateInput(io_1, p, io_2);
1000  }
1001  // Associate group's outputs to processor outputs
1002  for (List<Integer> list : ((Set<List<Integer>>) contents.get("output-associations")))
1003  {
1004  int io_1 = list.get(0);
1005  int io_2 = list.get(1);
1006  int p_id = list.get(2);
1007  if (!procs.containsKey(p_id))
1008  {
1009  // Ignore
1010  continue;
1011  }
1012  Processor p = procs.get(p_id);
1013  gp.associateOutput(io_1, p, io_2);
1014  }
1015  gp.m_notifySources = (Boolean) contents.get("notify-sources");
1016  return gp;
1017  }
