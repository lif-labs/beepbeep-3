153  @Override
154  protected boolean compute(Object[] inputs, Queue<Object[]> outputs)
155  {
156  int output_arity = getOutputArity();
157  Object[] f_value = new Object[1];
158  try
159  {
160  m_slicingFunction.evaluate(inputs, f_value);
161  }
162  catch (FunctionException e)
163  {
164  throw new ProcessorException(e);
165  }
166  Object slice_ids = f_value[0];
167  if (slice_ids == null)
168  {
169  // This event applies to no slice; don't bother processing it
170  return produceReturn(outputs);
171  }
172  Object[] slice_vals;
173  if (m_explodeArrays)
174  {
175  if (slice_ids.getClass().isArray())
176  {
177  slice_vals = (Object[]) slice_ids;
178  }
179  if (slice_ids instanceof Collection)
180  {
181  Collection<?> col = (Collection<?>) slice_ids;
182  slice_vals = new Object[col.size()];
183  int i = 0;
184  for (Object o : col)
185  {
186  slice_vals[i] = o;
187  i++;
188  }
189  }
190  else
191  {
192  slice_vals = new Object[] { slice_ids };
193  }
194  }
195  else
196  {
197  slice_vals = new Object[] { slice_ids };
198  }
199  for (Object slice_id : slice_vals)
200  {
201  Set<Object> slices_to_process = new HashSet<Object>();
202  if (slice_id instanceof ToAllSlices || slice_id == null)
203  {
204  slices_to_process.addAll(m_slices.keySet());
205  }
206  else
207  {
208  if (!m_slices.containsKey(slice_id))
209  {
210  // First time we see this value: create new slice
211  Processor p = m_processor.duplicate();
212  m_slices.put(slice_id, p);
213  addContextFromSlice(p, slice_id);
214  QueueSink sink = new QueueSink(output_arity);
215  Connector.connect(p, sink);
216  m_sinks.put(slice_id, sink);
217  if (m_eventTracker != null)
218  {
219  //p.setEventTracker(m_eventTracker.getCopy(false));
220  m_sliceIndices.put(slice_id, new ArrayList<Integer>());
221  }
222  }
223  slices_to_process.add(slice_id);
224  }
225  @SuppressWarnings("unchecked")
226  Future<Pushable>[] futures = new Future[slices_to_process.size()];
227  for (Object s_id : slices_to_process)
228  {
229  // Find processor corresponding to that slice
230  Processor slice_p = m_slices.get(s_id);
231  // If this slice hasn't been cleaned up...
232  if (slice_p != null)
233  {
234  QueueSink sink_p = m_sinks.get(s_id);
235  if (m_eventTracker != null)
236  {
237  m_sliceIndices.get(s_id).add(m_inputCount);
238  }
239  // Push the input into the processor
240  // Pushable[] p_array = new Pushable[inputs.length];
241  for (int i = 0; i < inputs.length; i++)
242  {
243  Object o_i = inputs[i];
244  Pushable p = slice_p.getPushableInput(i);
245  Future<Pushable> f = p.pushFast(o_i);
246  futures[i] = f;
247  // p_array[i] = p;
248  }
249  // Wait for all slices to be done
250  for (Future<Pushable> f : futures)
251  {
252  if (f != null)
253  {
254  try
255  {
256  f.get();
257  }
258  catch (InterruptedException e)
259  {
260  throw new ProcessorException(e);
261  }
262  catch (ExecutionException e)
263  {
264  throw new ProcessorException(e);
265  }
266  }
267  }
268  // Collect the output from that processor
269  Object[] out = sink_p.remove();
270  // Can we clean that slice?
271  Object[] can_clean = new Object[1];
272  if (m_cleaningFunction != null)
273  {
274  try
275  {
276  m_cleaningFunction.evaluate(out, can_clean);
277  }
278  catch (FunctionException e)
279  {
280  throw new ProcessorException(e);
281  }
282  }
283  if (can_clean != null && can_clean.length > 0 && can_clean[0] instanceof Boolean
284  && (Boolean) (can_clean[0]))
285  {
286  // Yes: remove the processor for that slice
287  m_slices.remove(s_id);
288  m_sinks.remove(s_id);
289  }
290  handleNewSliceValue(s_id, out[0], outputs);
291  }
292  }
293  }
294  m_inputCount++;
295  return produceReturn(outputs);
296  }
