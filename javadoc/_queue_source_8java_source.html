160  @Override
161  @SuppressWarnings("squid:S1168")
162  protected boolean compute(Object[] inputs, Queue<Object[]> outputs)
163  {
164  int size = m_events.size();
165  if (m_index >= size)
166  {
167  return false;
168  }
169  Object[] output = new Object[getOutputArity()];
170  Object event = m_events.get(m_index);
171  if (m_loop)
172  {
173  m_index = (m_index + 1) % size;
174  }
175  else
176  {
177  // If we don't loop, play the events only once
178  m_index++;
179  }
180  if (m_index > size && !m_loop)
181  {
182  // No more events from this queue
183  return false;
184  }
185  for (int i = 0; i < getOutputArity(); i++)
186  {
187  if (event == null)
188  {
189  // If one of the elements is null, don't output anything
190  return true;
191  }
192  output[i] = event;
193  }
194  outputs.add(output);
195  if (m_eventTracker != null)
196  {
197  int index = m_index - 1;
198  if (index < 0)
199  {
200  index += size;
201  }
202  for (int i = 0; i < getOutputArity(); i++)
203  {
204  associateTo(new QueueFunction(getId(), index), i, m_outputCount);
205  }
206  }
207  m_outputCount++;
208  return true;
209  }
