159  @Override
160  public synchronized Pushable push(Object o)
161  {
162  try
163  {
164  Queue<Object> q = m_inputQueues[m_index];
165  q.add(o);
166  }
167  catch (ArrayIndexOutOfBoundsException e)
168  {
169  throw new PushableException(e);
170  }
171  // Check if each input queue has an event ready
172  for (int i = 0; i < m_inputArity; i++)
173  {
174  Queue<Object> queue = m_inputQueues[i];
175  if (queue.isEmpty())
176  {
177  // One of them doesn't: we can't produce an output yet
178  return this;
179  }
180  }
181  // Pick an event from each input queue
182  Object[] inputs = new Object[m_inputArity];
183  for (int i = 0; i < m_inputArity; i++)
184  {
185  Queue<Object> queue = m_inputQueues[i];
186  Object ob = queue.remove();
187  inputs[i] = ob;
188  }
189  // Compute output event
190  m_tempQueue.clear();
191  boolean outs;
192  try
193  {
194  outs = compute(inputs, m_tempQueue);
195  }
196  catch (ProcessorException e)
197  {
198  throw new PushableException(e);
199  }
200  // put everything in outputEvent, because of duplicated code in notifyEndOfTrace
201  outputEvent(outs);
202  return this;
203  }

205  @Override
206  public void notifyEndOfTrace()
207  {
208  m_hasBeenNotifiedOfEndOfTrace[m_index] = true;
209  if (!allNotifiedEndOfTrace())
210  {
211  return;
212  }
213  m_tempQueue.clear();
214  boolean outs;
215  try
216  {
217  outs = onEndOfTrace(m_tempQueue);
218  }
219  catch (ProcessorException e)
220  {
221  throw new PushableException(e);
222  }
223  outputEvent(outs);
224 
225  // Notifies the output pushables of the end of the trace
226  for (int i = 0; i < m_outputPushables.length; i++)
227  {
228  Pushable p = m_outputPushables[i];
229  if (p == null)
230  {
231  throw new PushableException("Output " + i
232  + " of this processor is connected to nothing", getProcessor());
233  }
234  p.notifyEndOfTrace();
235  }
236  }

244  private final void outputEvent(boolean outs)
245  {
246  if (outs && !m_tempQueue.isEmpty())
247  {
248  for (Object[] evt : m_tempQueue)
249  {
250  if (evt != null)
251  {
252  for (int i = 0; i < m_outputPushables.length; i++)
253  {
254  Pushable p = m_outputPushables[i];
255  if (p == null)
256  {
257  throw new PushableException(
258  "Output " + i + " of processor " + SynchronousProcessor.this
259  + " is connected to nothing", getProcessor());
260  }
261  p.push(evt[i]);
262  }
263  }
264  }
265  m_tempQueue.clear();
266  }
267  }

355  @Override
356  public synchronized boolean hasNext()
357  {
358  Queue<Object> out_queue = m_outputQueues[m_index];
359  // If an event is already waiting in the output queue,
360  // return it and don't pull anything from the input
361  if (!out_queue.isEmpty())
362  {
363  return true;
364  }
365  // Check if each pullable has an event ready
366  for (int tries = 0; tries < Processor.MAX_PULL_RETRIES; tries++)
367  {
368  for (int i = 0; i < m_inputArity; i++)
369  {
370  Pullable p = m_inputPullables[i];
371  if (p == null)
372  {
373  throw new PullableException("Input " + i + " of processor " + SynchronousProcessor.this
374  + " is connected to nothing", getProcessor());
375  }
376  boolean status = p.hasNext();
377  if (!status)
378  {
379  if (allNotifiedEndOfTrace())
380  {
381  return false;
382  }
383  Queue<Object[]> last_queue = new ArrayDeque<Object[]>();
384  boolean b = onEndOfTrace(last_queue);
385  m_hasBeenNotifiedOfEndOfTrace[i] = true;
386  if (!b)
387  {
388  return false;
389  }
390  for (Object[] front : last_queue)
391  {
392  for (int j = 0; j < m_outputArity; j++)
393  {
394  m_outputQueues[j].add(front[j]);
395  }
396  }
397  return true;
398  }
399  }
400  // We are here only if every input pullable has answered YES
401  // Pull an event from each
402  Object[] inputs = new Object[m_inputArity];
403  for (int i = 0; i < m_inputArity; i++)
404  {
405  Pullable p = m_inputPullables[i];
406  // Don't check for p == null, we did it above
407  Object o = p.pull();
408  inputs[i] = o;
409  }
410  // Compute output event(s)
411  m_tempQueue.clear();
412  boolean computed;
413  try
414  {
415  computed = compute(inputs, m_tempQueue);
416  }
417  catch (ProcessorException e)
418  {
419  throw new PullableException(e);
420  }
421  NextStatus status_to_return = NextStatus.NO;
422  if (!computed)
423  {
424  // No output will ever be returned: stop there
425  return false;
426  }
427  if (!m_tempQueue.isEmpty())
428  {
429  // We computed an output event; add it to the output queue
430  // and answer YES
431  for (Object[] evt : m_tempQueue)
432  {
433  if (evt != null)
434  {
435  for (int i = 0; i < m_outputArity; i++)
436  {
437  Queue<Object> queue = m_outputQueues[i];
438  queue.add(evt[i]);
439  }
440  status_to_return = NextStatus.YES;
441  }
442  else
443  {
444  // This source will NEVER output anything again
445  m_tempQueue.clear();
446  return false;
447  }
448  }
449  if (status_to_return == NextStatus.YES)
450  {
451  m_tempQueue.clear();
452  return true;
453  }
454  }
455  // Otherwise, try the whole thing again
456  }
457  m_tempQueue.clear();
458  return false;
459  }

461  @Override
462  public synchronized NextStatus hasNextSoft()
463  {
464  Queue<Object> out_queue = m_outputQueues[m_index];
465  // If an event is already waiting in the output queue,
466  // return yes and don't pull anything from the input
467  if (!out_queue.isEmpty())
468  {
469  return NextStatus.YES;
470  }
471  // Check if each pullable has an event ready
472  for (int i = 0; i < m_inputArity; i++)
473  {
474  Pullable p = m_inputPullables[i];
475  NextStatus status = p.hasNextSoft();
476  if (status == NextStatus.NO)
477  {
478  if (allNotifiedEndOfTrace())
479  {
480  return NextStatus.NO;
481  }
482  Queue<Object[]> last_queue = new ArrayDeque<Object[]>();
483  boolean b = onEndOfTrace(last_queue);
484  m_hasBeenNotifiedOfEndOfTrace[i] = true;
485  if (!b)
486  {
487  return NextStatus.NO;
488  }
489  for (Object[] front : last_queue)
490  {
491  for (int j = 0; j < m_outputArity; j++)
492  {
493  m_outputQueues[j].add(front[j]);
494  }
495  }
496  return NextStatus.YES;
497  }
498  if (status == NextStatus.MAYBE)
499  {
500  return NextStatus.MAYBE;
501  }
502  }
503  // We are here only if every input pullable has answered YES
504  // Pull an event from each
505  Object[] inputs = new Object[m_inputArity];
506  {
507  int i = 0;
508  for (Pullable p : m_inputPullables)
509  {
510  inputs[i] = p.pullSoft();
511  i++;
512  }
513  }
514  // Compute output event(s)
515  NextStatus status_to_return = NextStatus.MAYBE;
516  m_tempQueue.clear();
517  boolean computed;
518  try
519  {
520  computed = compute(inputs, m_tempQueue);
521  }
522  catch (ProcessorException e)
523  {
524  throw new PullableException(e);
525  }
526  if (!computed)
527  {
528  status_to_return = NextStatus.NO;
529  }
530  else if (!m_tempQueue.isEmpty())
531  {
532  for (Object[] evt : m_tempQueue)
533  {
534  if (evt != null)
535  {
536  // We computed an output event; add it to the output queue
537  // and answer YES
538  int i = 0;
539  for (Queue<Object> queue : m_outputQueues)
540  {
541  queue.add(evt[i]);
542  i++;
543  }
544  status_to_return = NextStatus.YES;
545  }
546  }
547  }
548  return status_to_return;
549  }
