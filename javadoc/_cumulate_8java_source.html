50  @Override
51  protected boolean compute(Object[] inputs, Object[] outputs)
52  {
53  // We override compute() from ApplyFunction, only to complete the
54  // association between input and output events (each output event depends on
55  // the current input event and also the previous output event front)
56  boolean b = super.compute(inputs, outputs);
57  if (m_eventTracker != null)
58  {
59  for (int j = 0; j < outputs.length; j++)
60  {
61  if (m_outputCount > 1)
62  {
63  for (int k = 0; k < outputs.length; k++)
64  {
65  // The output event is associated to the current input event
66  // (this is already taken care of by super.compute)...
67  // ...and to the previous output event
68  // -1 and -2 since the count has already been incremented by the
69  // call to super.compute() above
70  associateToOutput(j, m_outputCount - 2, k, m_outputCount - 1);
71 
72  }
73  }
74  else
75  {
76  for (int k = 0; k < outputs.length; k++)
77  {
78  associateTo(new StartValue(getId(), j), k, m_outputCount - 1);
79  }
80  }
81  }
82  }
83  return b;
84  }
