89  @Override
90  @SuppressWarnings("squid:S3516")
91  protected boolean compute(Object[] inputs, Queue<Object[]> outputs)
92  {
93  // Add the inputs to each window
94  m_inputCount++;
95  boolean windows_ok = true;
96  int arity = inputs.length;
97  for (int i = 0; i < arity; i++)
98  {
99  LinkedList<Object> q = m_window[i];
100  q.add(inputs[i]);
101  int size_diff = q.size() - m_width;
102  leftTrim(size_diff, q);
103  if (size_diff < 0)
104  {
105  // Window is still to small to compute
106  windows_ok = false;
107  }
108  }
109  Object[] out = null;
110  if (windows_ok) // All windows have the proper width
111  {
112  m_processor.reset();
113  m_sink.reset();
114  int input_arity = getInputArity();
115  @SuppressWarnings("unchecked")
116  Future<Pushable>[] futures = new Future[m_width * input_arity];
117  for (int i = 0; i < m_width; i++)
118  {
119  for (int j = 0; j < input_arity; j++)
120  {
121  // Feed
122  LinkedList<Object> q = m_window[j];
123  Object o = q.get(i);
124  Pushable p = m_innerInputs[j];
125  futures[i * input_arity + j] = p.pushFast(o);
126  p.notifyEndOfTrace();
127  }
128  for (Future<?> f : futures)
129  {
130  if (f != null)
131  {
132  try
133  {
134  f.get();
135  }
136  catch (InterruptedException e)
137  {
138  throw new ProcessorException(e);
139  }
140  catch (ExecutionException e)
141  {
142  throw new ProcessorException(e);
143  }
144  }
145  }
146  out = m_sink.getLast();
147  }
148  }
149  if (out == null)
150  {
151  // Don't return false, otherwise it would signal that no
152  // event will every be produced in the future
153  return true;
154  }
155  else
156  {
157  m_outputCount++;
158  if (m_eventTracker != null)
159  {
160  for (int i = 1 - m_width; i <= 0; i++)
161  {
162  associateToInput(0, m_inputCount + i, 0, m_outputCount);
163  }
164  }
165  }
166  outputs.add(out);
167  return true;
168  }
