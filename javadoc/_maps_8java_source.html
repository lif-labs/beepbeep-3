290  {
294  protected Map<Object,Set<Object>> m_map;

299  public MergeMaps()
300  {
301  super(1, 1);
302  m_map = new HashMap<Object,Set<Object>>();
303  }
304 
305  @Override
306  protected boolean compute(Object[] inputs, Object[] outputs)
307  {
308  Map<?,?> m = (Map<?,?>) inputs[0];
309  for (Object m_o : m.entrySet())
310  {
311  if (m_o instanceof Map.Entry)
312  {
313  Map.Entry<?,?> e = (Map.Entry<?,?>) m_o;
314  Object key = e.getKey();
315  Object value = e.getValue();
316  if (value == null)
317  {
318  outputs[0] = m_map;
319  return true;
320  }
321  Set<Object> s_value;
322  if (m_map.containsKey(key))
323  {
324  s_value = m_map.get(key);
325  }
326  else
327  {
328  s_value = new HashSet<Object>();
329  }
330  if (value instanceof Collection)
331  {
332  s_value.addAll((Collection<?>) value);
333  }
334  else
335  {
336  s_value.add(value);
337  }
338  m_map.put(key, s_value);
339  }
340  }
341  outputs[0] = m_map;
342  return true;
343  }
344 
345  @Override
346  public void reset()
347  {
348  super.reset();
349  m_map.clear();
350  }
351 
352  @Override
353  public MergeMaps duplicate(boolean with_state)
354  {
355  MergeMaps mm = new MergeMaps();
356  if (with_state)
357  {
358  for (Map.Entry<Object,Set<Object>> e : m_map.entrySet())

361  new_set.addAll(e.getValue());
362  mm.m_map.put(e.getKey(), new_set);
363  }
